2. 変数と計算

            

#include <stdio.h>

int main(void) {
    int a = 10;
    int b = 20;
    int sum = a + b;
    
    printf("a = %d, b = %d\n", a, b);
    printf("a + b = %d\n", sum);
    return 0;
}
